\begin{frame}{DWARF CFI (Call Frame Information): CFA}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{onlyenv}<1>
      \begin{itemize}
      \item DWARF\footnote{\url{https://dwarfstd.org/doc/DWARF5.pdf}}, {\S}6.4, specifies how to encode compressed form: \underline{Call Frame Information}

      \item Stack frames are identified by their \alert{CFA} (Canonical Frame Address)
        \begin{itemize}
        \item Typically, \texttt{SP} at the call site in the prev. frame
          \includegraphics[width=\columnwidth]{img/may_throw-armv8a-bl.png}
        \end{itemize}
      \end{itemize}
    \end{onlyenv}

\begin{frame}{DWARF CFI (Call Frame Information): CFI instructions (1)}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\columnwidth]{img/may_throw-armv8a-prologue.png}
    \end{center}

    \column{0.5\textwidth}
    Usually, CFI instructions (opcodes) specify\ldots{}
    \begin{itemize}
    \item An initial CFA rule (after prologue)\\
      $\rightarrow$~\inlineCode{DW\_CFA\_def\_cfa\_offset}
    \item CFA-relative position of callee-saved, \texttt{FP}, and return registers (\texttt{LR})\\
      $\rightarrow$~\inlineCode{DW\_CFA\_offset}
      \bigskip

    \item And\ldots{} corrections of the CFA\\
      $\rightarrow$~\inlineCode{DW\_CFA\_def\_cfa}\\
      $\rightarrow$~\inlineCode{DW\_CFA\_def\_cfa\_offset}\\
      $\rightarrow$~\inlineCode{DW\_CFA\_advance\_loc}
    \end{itemize}
  \end{columns}

  \begin{onlyenv}<2->
    \overlayLayer{\texttt{FP}-based CFA rule is preferred!}
  \end{onlyenv}
\end{frame}

\begin{frame}{\texttt{.eh\_frame}: CIEs / FDEs}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}
    \item A collection of ``CFI'' records:
      \begin{itemize}
      \item CIE (Common Information Entry)
      \item FDE (Frame Description Entry)
      \end{itemize}
      \bigskip

    \item Any number of FDEs can follow a CIE
    \end{itemize}

    \column{0.5\textwidth}
    \begin{onlyenv}<1>
      \resizebox{\columnwidth}{!}{\input{fig/cie-fde-hier}}
    \end{onlyenv}

    \begin{onlyenv}<2>
      \underline{CIE (Common Information Entry)}\\[1ex]
      \begin{itemize}
      \item Code and Data Alignment factors
      \item Return address (DWARF) register
      \item \alert{Augmentation string} [+ data + length]
        \begin{itemize}
        \item \inlineCode{z}: has data
        \item \inlineCode{L}: has LSDA
        \item \inlineCode{P}: has personality
        \end{itemize}

      \item Initial CFI `program'
      \end{itemize}
    \end{onlyenv}

    \begin{onlyenv}<3>
      \underline{FDE (Frame Description Entry)}\\[1ex]
      \begin{itemize}
      \item PC range
      \item Augmentation data
      \item CFI `program`
      \end{itemize}
    \end{onlyenv}
  \end{columns}

  \begin{onlyenv}<4>
    \overlayLayer{\resizebox{0.54\textwidth}{!}{\input{fig/cie-fde}}}
  \end{onlyenv}
\end{frame}

\begin{frame}{Compact Unwind: \texttt{\_\_unwind\_info} (macOS / iOS only)}
  \alert{Very efficient} encoding of unwind information (given some assumptions)!
  %
  \vfill
  \begin{columns}
    \column{0.5\textwidth}
    \underline{\bfseries Good:}\\

    \column{0.5\textwidth}
    \underline{\bfseries Bad:}\\
  \end{columns}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}
    \item Can encode a range's unwind-info in 32-bits (most of the time)
    \item Replaces both \texttt{.eh\_frame} and \texttt{.eh\_frame\_hdr}
    \end{itemize}

    \column{0.5\textwidth}
    \begin{itemize}
    \item Only for Apple platforms
    \end{itemize}
  \end{columns}
\end{frame}
